// COMMON FUNCTION
function modExp(base, exp, mod){
    let result = 1;
    base = base % mod;

    while(exp > 0){
        if(exp % 2 === 1){
            result = (result * base) % mod;
        }
        base = (base * base) % mod;
        exp = Math.floor(exp / 2);
    }
    return result;
}

// ================= Diffie-Hellman =================
function calculateDH(){

    let p = parseInt(document.getElementById("p").value);
    let g = parseInt(document.getElementById("g").value);
    let a = parseInt(document.getElementById("a").value);
    let b = parseInt(document.getElementById("b").value);
    let out = document.getElementById("outputDH");

    if(isNaN(p) || isNaN(g) || isNaN(a) || isNaN(b)){
        out.innerHTML = "Enter valid numbers";
        return;
    }

    let A = modExp(g, a, p);
    let B = modExp(g, b, p);

    let keyA = modExp(B, a, p);
    let keyB = modExp(A, b, p);

    out.innerHTML = `
    <b>Step 1: Public Keys</b><br>
    A = ${A}<br>
    B = ${B}<br><br>

    <b>Step 2: Shared Secret</b><br>
    Key (A) = ${keyA}<br>
    Key (B) = ${keyB}<br><br>

    <b>Final Shared Key = ${keyA}</b>
    `;
}

// ================= RSA =================
function gcd(a,b){
    return b==0 ? a : gcd(b, a%b);
}

function modInverse(e, phi){
    for(let d=1; d<phi; d++){
        if((e*d) % phi === 1){
            return d;
        }
    }
    return null;
}

function calculateRSA(){

    let p = parseInt(document.getElementById("rp").value);
    let q = parseInt(document.getElementById("rq").value);
    let e = parseInt(document.getElementById("re").value);
    let m = parseInt(document.getElementById("msg").value);

    let out = document.getElementById("outputRSA");

    if(isNaN(p) || isNaN(q) || isNaN(e) || isNaN(m)){
        out.innerHTML = " Enter valid numbers";
        return;
    }

    let n = p * q;
    let phi = (p-1)*(q-1);

    if(gcd(e, phi) !== 1){
        out.innerHTML = " e must be coprime with φ(n)";
        return;
    }

    let d = modInverse(e, phi);

    let cipher = modExp(m, e, n);
    let decrypted = modExp(cipher, d, n);

    out.innerHTML = `
    <div class="label">Step 1: Key Generation</div><br>
    n = ${n}<br>
    φ(n) = ${phi}<br>
    Public Key = (${e}, ${n})<br>
    Private Key = (${d}, ${n})<br><br>

    <div class="label">Step 2: Encryption</div><br>
    C = ${cipher}<br><br>

    <div class="label">Step 3: Decryption</div><br>
    M = ${decrypted}<br><br>

    <div class="label">Final Decrypted Message: ${decrypted}</div>
    `;
}
