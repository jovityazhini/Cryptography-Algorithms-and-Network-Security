// ================= S-DES =================
document.getElementById("btn").addEventListener("click", function() {

    let key = document.getElementById("key").value.trim();
    let pt = document.getElementById("plaintext").value.trim();
    let resultDiv = document.getElementById("result");

    if (!/^[01]{10}$/.test(key)) {
        resultDiv.innerHTML = " Enter 10-bit binary key";
        return;
    }

    if (!/^[01]{8}$/.test(pt)) {
        resultDiv.innerHTML = " Enter 8-bit binary plaintext";
        return;
    }

    function permute(bits, table) {
        return table.map(i => bits[i - 1]).join('');
    }

    function leftShift(bits, n) {
        return bits.slice(n) + bits.slice(0, n);
    }

    function xor(a, b) {
        return a.split('').map((bit, i) => bit ^ b[i]).join('');
    }

    const S0 = [
        ["01","00","11","10"],
        ["11","10","01","00"],
        ["00","10","01","11"],
        ["11","01","11","10"]
    ];

    const S1 = [
        ["00","01","10","11"],
        ["10","00","01","11"],
        ["11","00","01","00"],
        ["10","01","00","11"]
    ];

    function sbox(input, box) {
        let row = parseInt(input[0] + input[3], 2);
        let col = parseInt(input[1] + input[2], 2);
        return box[row][col];
    }

    let P10 = [5,3,8,6,10,2,9,1,4,7];
    let P8 = [6,4,7,3,9,10,5,1];

    let p10 = permute(key, P10);
    let left = p10.slice(0,5);
    let right = p10.slice(5);

    left = leftShift(left,1);
    right = leftShift(right,1);
    let k1 = permute(left + right, P8);

    left = leftShift(left,2);
    right = leftShift(right,2);
    let k2 = permute(left + right, P8);

    let IP = [3,5,2,7,4,1,8,6];
    let EP = [4,1,2,3,2,3,4,1];
    let P4 = [2,4,3,1];
    let IP_inv = [6,3,1,5,2,8,4,7];

    let ip = permute(pt, IP);
    let L = ip.slice(0,4);
    let R = ip.slice(4);

    let ep = permute(R, EP);
    let xor1 = xor(ep, k1);

    let leftS = sbox(xor1.slice(0,4), S0);
    let rightS = sbox(xor1.slice(4), S1);

    let p4 = permute(leftS + rightS, P4);
    let L1 = xor(L, p4);

    let swapped = R + L1;

    let L2 = swapped.slice(0,4);
    let R2 = swapped.slice(4);

    let ep2 = permute(R2, EP);
    let xor2 = xor(ep2, k2);

    let leftS2 = sbox(xor2.slice(0,4), S0);
    let rightS2 = sbox(xor2.slice(4), S1);

    let p4_2 = permute(leftS2 + rightS2, P4);
    let L3 = xor(L2, p4_2);

    let preOutput = L3 + R2;
    let cipher = permute(preOutput, IP_inv);

    resultDiv.innerHTML = " Cipher Text: " + cipher;
});

const RCON = ["01","02","03","08"];

function xorHex(a,b){
    return (parseInt(a,16) ^ parseInt(b,16)).toString(16).padStart(2,'0');
}

function subWord(word){
    return word.map(b => SBOX[parseInt(b,16)]);
}

function rotWord(word){
    return [word[1], word[2], word[3], word[0]];
}

function generate(){

    let key = document.getElementById("aeskey").value.trim();
    let out = document.getElementById("output");

    if(!/^[0-9a-fA-F]{32}$/.test(key)){
        out.innerText = " Enter valid 32 hex characters";
        return;
    }

    let words = [];

    for(let i=0;i<4;i++){
        words[i] = key.slice(i*8, (i+1)*8).match(/../g);
    }

    let result = " Round 0 Key:\n" + words.map(w=>w.join(" ")).join("\n") + "\n\n";

    for(let i=4;i<20;i++){
        let temp = [...words[i-1]];

        if(i%4==0){
            temp = rotWord(temp);
            temp = subWord(temp);
            temp[0] = xorHex(temp[0], RCON[(i/4)-1]);
        }

        let newWord = words[i-4].map((b,j)=> xorHex(b, temp[j]));
        words[i] = newWord;

        if(i%4==3){
            let round =Math.floor(i/4);
            result += " Round " + round + " Key:\n";
            result += words[i-3].join(" ") + "\n";
            result += words[i-2].join(" ") + "\n";
            result += words[i-1].join(" ") + "\n";
            result += words[i].join(" ") + "\n\n";
        }
    }

    out.innerText = result;
}
