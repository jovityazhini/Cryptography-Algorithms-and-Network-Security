<script>
    // UI Logic
    function showTab(tabId) {
        document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
        document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
        document.getElementById(tabId).classList.add('active');
        event.currentTarget.classList.add('active');
    }

    function display(msg, isError = false) {
        const el = document.getElementById('resultText');
        el.innerHTML = msg;
        el.className = isError ? 'error' : '';
    }


    const mod = (n, m) => ((n % m) + m) % m;

    //  1. Shift Cipher
    function runShift(encrypt) {
        let text = document.getElementById('shiftText').value;
        let k = parseInt(document.getElementById('shiftKey').value);
        if (!encrypt) k = -k;

        let res = text.replace(/[a-z]/gi, (char) => {
            let base = (char <= 'Z') ? 65 : 97;
            return String.fromCharCode(mod(char.charCodeAt(0) - base + k, 26) + base);
        });
        display(res);
    }

    // 2. Hill Cipher (2x2)
    function getModInverse(a, m) {
        a = mod(a, m);
        for (let x = 1; x < m; x++) {
            if ((a * x) % m === 1) return x;
        }
        return null;
    }

    function runHill(encrypt) {
        let text = document.getElementById('hillText').value.toUpperCase().replace(/[^A-Z]/g, '');
        let keyStr = document.getElementById('hillKey').value.toUpperCase().replace(/[^A-Z]/g, '');

        if (keyStr.length !== 4) return display("Error: Key must be 4 characters.", true);
        if (text.length % 2 !== 0) text += 'X';

        let k = keyStr.split('').map(c => c.charCodeAt(0) - 65);
        let a = k[0], b = k[1], c = k[2], d = k[3];
        let matrix;

        if (encrypt) {
            matrix = [a, b, c, d];
        } else {
            let det = mod(a * d - b * c, 26);
            let invDet = getModInverse(det, 26);
            if (invDet === null) return display(`Error: Key is not invertible (Det: ${det}). Try another key.`, true);
            matrix = [mod(d * invDet, 26), mod(-b * invDet, 26), mod(-c * invDet, 26), mod(a * invDet, 26)];
        }

        let res = "";
        for (let i = 0; i < text.length; i += 2) {
            let p1 = text.charCodeAt(i) - 65;
            let p2 = text.charCodeAt(i + 1) - 65;
            res += String.fromCharCode(mod(matrix[0] * p1 + matrix[1] * p2, 26) + 65);
            res += String.fromCharCode(mod(matrix[2] * p1 + matrix[3] * p2, 26) + 65);
        }
        display(res);
    }

    // 3. Playfair Cipher
    function runPlayfair(encrypt) {
        let text = document.getElementById('playfairText').value.toUpperCase().replace(/J/g, 'I').replace(/[^A-Z]/g, '');
        let key = document.getElementById('playfairKey').value.toUpperCase().replace(/J/g, 'I').replace(/[^A-Z]/g, '');

        let alphabet = "ABCDEFGHIKLMNOPQRSTUVWXYZ";
        let gridChars = [...new Set((key + alphabet).split(''))];
        let matrix = [], pos = {};

        gridChars.forEach((char, i) => {
            let r = Math.floor(i / 5), c = i % 5;
            if (!matrix[r]) matrix[r] = [];
            matrix[r][c] = char;
            pos[char] = [r, c];
        });

        let pairs = [];
        for (let i = 0; i < text.length; i += 2) {
            let a = text[i], b = text[i+1] || 'X';
            if (a === b && encrypt) { b = 'X'; i--; }
            pairs.push([a, b]);
        }

        let shift = encrypt ? 1 : -1;
        let res = pairs.map(p => {
            let [r1, c1] = pos[p[0]], [r2, c2] = pos[p[1]];
            if (r1 === r2) return matrix[r1][mod(c1 + shift, 5)] + matrix[r2][mod(c2 + shift, 5)];
            if (c1 === c2) return matrix[mod(r1 + shift, 5)][c1] + matrix[mod(r2 + shift, 5)][c2];
            return matrix[r1][c2] + matrix[r2][c1];
        }).join('');
        display(res);
    }

    //  4. Miller-Rabin Primality
    function bigPower(base, exp, mod) {
        let res = 1n; base = BigInt(base) % BigInt(mod);
        while (exp > 0n) {
            if (exp % 2n === 1n) res = (res * base) % BigInt(mod);
            base = (base * base) % BigInt(mod); exp = exp / 2n;
        }
        return res;
    }

    function runMiller() {
        let n = BigInt(document.getElementById('millerNum').value || 0);
        let k = parseInt(document.getElementById('millerK').value || 5);

        if (n < 2n) return display("Not Prime", true);
        if (n === 2n || n === 3n) return display("Prime");
        if (n % 2n === 0n) return display("Composite (Even number)");

        let d = n - 1n, s = 0n;
        while (d % 2n === 0n) { d /= 2n; s++; }

        for (let i = 0; i < k; i++) {
            let a = 2n + BigInt(Math.floor(Math.random() * Number(n - 4n)));
            let x = bigPower(a, d, n);
            if (x === 1n || x === n - 1n) continue;
            let composite = true;
            for (let r = 1n; r < s; r++) {
                x = (x * x) % n;
                if (x === n - 1n) { composite = false; break; }
            }
            if (composite) return display("Result: Composite");
        }
        display("Result: Probably Prime");
    }

    // 5. Euclidean Algorithm
    function runEuclidean() {
        let a = Math.abs(parseInt(document.getElementById('eucA').value || 0));
        let b = Math.abs(parseInt(document.getElementById('eucB').value || 0));
        let steps = [];
        let originalA = a, originalB = b;

        while (b) {
            steps.push(`${a} = (${Math.floor(a/b)} &times; ${b}) + ${a % b}`);
            let t = b; b = a % b; a = t;
        }
        display(`GCD(${originalA}, ${originalB}) = <strong>${a}</strong><br><br><small>${steps.join('<br>')}</small>`);
    }
